#include <GL/glut.h>    
 
int lookToRight, 
	lookToLeft,
	lookToBottom,
	lookToTop,
	distance;
float zoom;

void reshape(int w, int h)
{
        glViewport(0, 0, w, h);
       
        glMatrixMode(GL_PROJECTION);
        glLoadIdentity();
        //gluOrtho2D(0, w, 0, h);
		gluPerspective(60, w/h, 0, 1000);
		
        glMatrixMode(GL_MODELVIEW);
        glLoadIdentity();

		gluLookAt(200, 200, distance,
			0, 0, 0,
			0, 50, 0);
		
}

void init(){
	distance = 600;
	zoom = 0;
}

void display()
{
        glClear(GL_COLOR_BUFFER_BIT);
		if(lookToRight){
			glRotatef(1, 0.0f, 1.0f, 0.0f);
			lookToRight = 0;
		} 
		if (lookToLeft){
			glRotatef(-1, 0.0f, 1.0f, 0.0f);
			lookToLeft = 0;
		}
		if (lookToTop){
			glRotatef(1, 1.0f, 0.0f, 0.0f);
			lookToTop = 0;
		}
		if (lookToBottom){
			glRotatef(-1, 1.0f, 0.0f, 0.0f);
			lookToBottom = 0;
		}

		if(zoom != 0){
			glScalef(zoom,zoom,zoom);
			zoom = 0;
		}
		
       
        glBegin(GL_POLYGON);
        glColor3f(1.0, 0, 0);
        glVertex3i(0, 0, 0);

        glColor3f(0.0, 1.0, 1.0);
        glVertex3i(100, 0, 0);
        
		glColor3f(0.0, 1.0, 0.0);
        glVertex3i(100, 100, 100);
        
		glColor3f(0, 0, 1.0);
        glVertex3i(0, 100, 0);
        glEnd();


    	glBegin(GL_LINES);

		glColor3f(0, 1, 0); /* OY green*/
		glVertex3i(0, 0, 0);
		glVertex3i(0, 500, 0);	

		glColor3f(0, 0, 1); /* OX blue*/
		glVertex3i(0, 0, 0);
		glVertex3i(500, 0, 0);	

		glColor3f(1, 0, 0); /* OZ red*/
		glVertex3i(0, 0, 0);
		glVertex3i(0, 0, 500);	
		
		glEnd();

		glBegin(GL_LINES);
		glColor3f(0.5, 0.5, 0.5);
		for (int i=1; i<=10; ++i){
			glColor3f(0, 0, 0.5); /* от OX линии*/
			glVertex3i(i*50, 0, 0);
			glColor3f(1, 1, 1);
			glVertex3i(i*50, 0, 500);

		}
		glEnd();

        glutSwapBuffers();
}

void keyboard (unsigned char key, int x, int y) {
	if (key == 97 /* A */)
		lookToLeft = 1;
	if (key == 100 /* D */)
		lookToRight = 1;
	if (key == 119 /* W */)
		lookToBottom = 1;
	if (key == 115 /* S */)
		lookToTop = 1;
	if (key == 43 /* + */)
		zoom = 2;
	if (key == 45 /* - */){
		zoom = 0.5;
	}

	
	
	//glMatrixMode(GL_MODELVIEW);
    //glLoadIdentity();

	/*gluLookAt(eyeX, eyeY, 1000,
			0, 0, 0,
			0, 50, 0);
	*/
	
	display();
}


int main (int argc, char * argv[])
{
	init();
    glutInit(&argc, argv);
    glutInitDisplayMode(GLUT_DOUBLE|GLUT_RGBA);
       
    glutInitWindowSize(200, 200);
    glutCreateWindow("OpenGL lesson 1");
       
    glutReshapeFunc(reshape);
    glutDisplayFunc(display);
    glutKeyboardFunc (keyboard); 
        
	glutMainLoop();
       
    return 0;
}
